109 // state estimation callback function
110 void odometryHandler(const nav_msgs::msg::Odometry::ConstSharedPtr odom) {
111  double roll, pitch, yaw;
112  geometry_msgs::msg::Quaternion geoQuat = odom->pose.pose.orientation;
113  tf2::Matrix3x3(tf2::Quaternion(geoQuat.x, geoQuat.y, geoQuat.z, geoQuat.w))
114  .getRPY(roll, pitch, yaw);
115 
116  vehicleRoll = roll;
117  vehiclePitch = pitch;
118  vehicleYaw = yaw;
119  vehicleX = odom->pose.pose.position.x;
120  vehicleY = odom->pose.pose.position.y;
121  vehicleZ = odom->pose.pose.position.z;
122 
123  sinVehicleRoll = sin(vehicleRoll);
124  cosVehicleRoll = cos(vehicleRoll);
125  sinVehiclePitch = sin(vehiclePitch);
126  cosVehiclePitch = cos(vehiclePitch);
127  sinVehicleYaw = sin(vehicleYaw);
128  cosVehicleYaw = cos(vehicleYaw);
129 
130  if (noDataInited == 0) {
131  vehicleXRec = vehicleX;
132  vehicleYRec = vehicleY;
133  noDataInited = 1;
134  }
135  if (noDataInited == 1) {
136  float dis = sqrt((vehicleX - vehicleXRec) * (vehicleX - vehicleXRec) +
137  (vehicleY - vehicleYRec) * (vehicleY - vehicleYRec));
138  if (dis >= noDecayDis)
139  noDataInited = 2;
140  }
141 }

143 // registered laser scan callback function
144 void laserCloudHandler(
145  const sensor_msgs::msg::PointCloud2::ConstSharedPtr laserCloud2) {
146  laserCloudTime = rclcpp::Time(laserCloud2->header.stamp).seconds();
147  if (!systemInited) {
148  systemInitTime = laserCloudTime;
149  systemInited = true;
150  }
151 
152  laserCloud->clear();
153  pcl::fromROSMsg(*laserCloud2, *laserCloud);
154 
155  pcl::PointXYZI point;
156  laserCloudCrop->clear();
157  int laserCloudSize = laserCloud->points.size();
158  for (int i = 0; i < laserCloudSize; i++) {
159  point = laserCloud->points[i];
160 
161  float pointX = point.x;
162  float pointY = point.y;
163  float pointZ = point.z;
164 
165  float dis = sqrt((pointX - vehicleX) * (pointX - vehicleX) +
166  (pointY - vehicleY) * (pointY - vehicleY));
167  if (pointZ - vehicleZ > minRelZ - disRatioZ * dis &&
168  pointZ - vehicleZ < maxRelZ + disRatioZ * dis &&
169  dis < terrainVoxelSize * (terrainVoxelHalfWidth + 1)) {
170  point.x = pointX;
171  point.y = pointY;
172  point.z = pointZ;
173  point.intensity = laserCloudTime - systemInitTime;
174  laserCloudCrop->push_back(point);
175  }
176  }
177 
178  newlaserCloud = true;
179 }
