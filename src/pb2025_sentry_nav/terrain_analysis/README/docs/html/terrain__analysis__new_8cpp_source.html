192 TerrainAnalysisNode::void odometryHandler(
193  const nav_msgs::msg::Odometry::ConstSharedPtr odom)
194 {
195  double roll, pitch, yaw;
196  geometry_msgs::msg::Quaternion geoQuat = odom->pose.pose.orientation;
197  tf2::Matrix3x3(tf2::Quaternion(geoQuat.x, geoQuat.y, geoQuat.z, geoQuat.w))
198  .getRPY(roll, pitch, yaw);
199 
200  vehicleRoll = roll;
201  vehiclePitch = pitch;
202  vehicleYaw = yaw;
203  vehicleX = odom->pose.pose.position.x;
204  vehicleY = odom->pose.pose.position.y;
205  vehicleZ = odom->pose.pose.position.z;
206 
207  sinVehicleRoll = sin(vehicleRoll);
208  cosVehicleRoll = cos(vehicleRoll);
209  sinVehiclePitch = sin(vehiclePitch);
210  cosVehiclePitch = cos(vehiclePitch);
211  sinVehicleYaw = sin(vehicleYaw);
212  cosVehicleYaw = cos(vehicleYaw);
213 
214  if (no_data_inited_ == 0)
215  {
216  vehicleXRec = vehicleX;
217  vehicleYRec = vehicleY;
218  no_data_inited_ = 1;
219  }
220  if (no_data_inited_ == 1)
221  {
222  float dis = sqrt((vehicleX - vehicleXRec) * (vehicleX - vehicleXRec) +
223  (vehicleY - vehicleYRec) * (vehicleY - vehicleYRec));
224  if (dis >= no_decay_dis_)
225  no_data_inited_ = 2;
226  }
227 }

229 TerrainAnalysisNode::void laserCloudHandler(
230  const sensor_msgs::msg::PointCloud2::ConstSharedPtr laserCloud2)
231 {
232  laser_cloud_time_ = rclcpp::Time(laserCloud2->header.stamp).seconds();
233  if (!system_inited_)
234  {
235  system_init_time_ = laser_cloud_time_;
236  system_inited_ = true;
237  }
238 
239  laserCloud->clear();
240  pcl::fromROSMsg(*laserCloud2, *laserCloud);
241 
242  pcl::PointXYZI point;
243  laserCloudCrop->clear();
244  int laserCloudSize = laserCloud->points.size();
245  for (int i = 0; i < laserCloudSize; i++)
246  {
247  point = laserCloud->points[i];
248 
249  float pointX = point.x;
250  float pointY = point.y;
251  float pointZ = point.z;
252 
253  float dis = sqrt((pointX - vehicleX) * (pointX - vehicleX) +
254  (pointY - vehicleY) * (pointY - vehicleY));
255  if (pointZ - vehicleZ > min_rel_z_ - dis_ratio_z_ * dis &&
256  pointZ - vehicleZ < max_rel_z_ + dis_ratio_z_ * dis &&
257  dis < terrainVoxelSize * (terrainVoxelHalfWidth + 1))
258  {
259  point.x = pointX;
260  point.y = pointY;
261  point.z = pointZ;
262  point.intensity = laser_cloud_time_ - system_init_time_;
263  laserCloudCrop->push_back(point);
264  }
265  }
266 
267  new_laser_cloud_ = true;
268 }

287 TerrainAnalysisNode::void timerCallback()
288 {
289  if (new_laser_cloud_)
290  {
291  new_laser_cloud_ = false;
292 
293  // terrain voxel roll over
294  float terrainVoxelCenX = terrainVoxelSize * terrain_voxel_shift_x_;
295  float terrainVoxelCenY = terrainVoxelSize * terrain_voxel_shift_y_;
296 
297  while (vehicleX - terrainVoxelCenX < -terrainVoxelSize)
298  {
299  for (int indY = 0; indY < terrainVoxelWidth; indY++)
300  {
301  pcl::PointCloud<pcl::PointXYZI>::Ptr terrainVoxelCloudPtr =
302  terrainVoxelCloud[terrainVoxelWidth * (terrainVoxelWidth - 1) +
303  indY];
304  for (int indX = terrainVoxelWidth - 1; indX >= 1; indX--)
305  {
306  terrainVoxelCloud[terrainVoxelWidth * indX + indY] =
307  terrainVoxelCloud[terrainVoxelWidth * (indX - 1) + indY];
308  }
309  terrainVoxelCloud[indY] = terrainVoxelCloudPtr;
310  terrainVoxelCloud[indY]->clear();
311  }
312  terrain_voxel_shift_x_--;
313  terrainVoxelCenX = terrainVoxelSize * terrain_voxel_shift_x_;
314  }
315 
316  while (vehicleX - terrainVoxelCenX > terrainVoxelSize)
317  {
318  for (int indY = 0; indY < terrainVoxelWidth; indY++)
319  {
320  pcl::PointCloud<pcl::PointXYZI>::Ptr terrainVoxelCloudPtr =
321  terrainVoxelCloud[indY];
322  for (int indX = 0; indX < terrainVoxelWidth - 1; indX++)
323  {
324  terrainVoxelCloud[terrainVoxelWidth * indX + indY] =
325  terrainVoxelCloud[terrainVoxelWidth * (indX + 1) + indY];
326  }
327  terrainVoxelCloud[terrainVoxelWidth * (terrainVoxelWidth - 1) +
328  indY] = terrainVoxelCloudPtr;
329  terrainVoxelCloud[terrainVoxelWidth * (terrainVoxelWidth - 1) + indY]
330  ->clear();
331  }
332  terrain_voxel_shift_x_++;
333  terrainVoxelCenX = terrainVoxelSize * terrain_voxel_shift_x_;
334  }
335 
336  while (vehicleY - terrainVoxelCenY < -terrainVoxelSize)
337  {
338  for (int indX = 0; indX < terrainVoxelWidth; indX++)
339  {
340  pcl::PointCloud<pcl::PointXYZI>::Ptr terrainVoxelCloudPtr =
341  terrainVoxelCloud[terrainVoxelWidth * indX +
342  (terrainVoxelWidth - 1)];
343  for (int indY = terrainVoxelWidth - 1; indY >= 1; indY--)
344  {
345  terrainVoxelCloud[terrainVoxelWidth * indX + indY] =
346  terrainVoxelCloud[terrainVoxelWidth * indX + (indY - 1)];
347  }
348  terrainVoxelCloud[terrainVoxelWidth * indX] = terrainVoxelCloudPtr;
349  terrainVoxelCloud[terrainVoxelWidth * indX]->clear();
350  }
351  terrain_voxel_shift_y_--;
352  terrainVoxelCenY = terrainVoxelSize * terrain_voxel_shift_y_;
353  }
354 
355  while (vehicleY - terrainVoxelCenY > terrainVoxelSize)
356  {
357  for (int indX = 0; indX < terrainVoxelWidth; indX++)
358  {
359  pcl::PointCloud<pcl::PointXYZI>::Ptr terrainVoxelCloudPtr =
360  terrainVoxelCloud[terrainVoxelWidth * indX];
361  for (int indY = 0; indY < terrainVoxelWidth - 1; indY++)
362  {
363  terrainVoxelCloud[terrainVoxelWidth * indX + indY] =
364  terrainVoxelCloud[terrainVoxelWidth * indX + (indY + 1)];
365  }
366  terrainVoxelCloud[terrainVoxelWidth * indX +
367  (terrainVoxelWidth - 1)] = terrainVoxelCloudPtr;
368  terrainVoxelCloud[terrainVoxelWidth * indX + (terrainVoxelWidth - 1)]
369  ->clear();
370  }
371  terrain_voxel_shift_y_++;
372  terrainVoxelCenY = terrainVoxelSize * terrain_voxel_shift_y_;
373  }
374 
375  // stack registered laser scans
376  pcl::PointXYZI point;
377  int laserCloudCropSize = laserCloudCrop->points.size();
378  for (int i = 0; i < laserCloudCropSize; i++)
379  {
380  point = laserCloudCrop->points[i];
381 
382  int indX =
383  static_cast<int>((point.x - vehicleX + terrainVoxelSize / 2) /
384  terrainVoxelSize) +
385  terrainVoxelHalfWidth;
386  int indY =
387  static_cast<int>((point.y - vehicleY + terrainVoxelSize / 2) /
388  terrainVoxelSize) +
389  terrainVoxelHalfWidth;
390 
391  if (point.x - vehicleX + terrainVoxelSize / 2 < 0)
392  indX--;
393  if (point.y - vehicleY + terrainVoxelSize / 2 < 0)
394  indY--;
395 
396  if (indX >= 0 && indX < terrainVoxelWidth && indY >= 0 &&
397  indY < terrainVoxelWidth)
398  {
399  terrainVoxelCloud[terrainVoxelWidth * indX + indY]->push_back(point);
400  terrainVoxelUpdateNum[terrainVoxelWidth * indX + indY]++;
401  }
402  }
403 
404  for (int ind = 0; ind < kTerrainVoxelNum; ind++)
405  {
406  if (terrainVoxelUpdateNum[ind] >= voxel_point_update_thre_ ||
407  laser_cloud_time_ - system_init_time_ - terrainVoxelUpdateTime[ind] >=
408  voxel_time_update_thre_ ||
409  clearing_cloud_)
410  {
411  pcl::PointCloud<pcl::PointXYZI>::Ptr terrainVoxelCloudPtr =
412  terrainVoxelCloud[ind];
413 
414  laserCloudDwz->clear();
415  downSizeFilter.setInputCloud(terrainVoxelCloudPtr);
416  downSizeFilter.filter(*laserCloudDwz);
417 
418  terrainVoxelCloudPtr->clear();
419  int laserCloudDwzSize = laserCloudDwz->points.size();
420  for (int i = 0; i < laserCloudDwzSize; i++)
421  {
422  point = laserCloudDwz->points[i];
423  float dis = sqrt((point.x - vehicleX) * (point.x - vehicleX) +
424  (point.y - vehicleY) * (point.y - vehicleY));
425  if (point.z - vehicleZ > min_rel_z_ - dis_ratio_z_ * dis &&
426  point.z - vehicleZ < max_rel_z_ + dis_ratio_z_ * dis &&
427  (laser_cloud_time_ - system_init_time_ - point.intensity <
428  decay_time_ ||
429  dis < no_decay_dis_) &&
430  !(dis < clearing_dis_ && clearing_cloud_))
431  {
432  terrainVoxelCloudPtr->push_back(point);
433  }
434  }
435 
436  terrainVoxelUpdateNum[ind] = 0;
437  terrainVoxelUpdateTime[ind] = laser_cloud_time_ - system_init_time_;
438  }
439  }
440 
441  terrainCloud->clear();
442  for (int indX = terrainVoxelHalfWidth - 5;
443  indX <= terrainVoxelHalfWidth + 5; indX++)
444  {
445  for (int indY = terrainVoxelHalfWidth - 5;
446  indY <= terrainVoxelHalfWidth + 5; indY++)
447  {
448  *terrainCloud += *terrainVoxelCloud[terrainVoxelWidth * indX + indY];
449  }
450  }
451 
452  // estimate ground and compute elevation for each point
453  for (int i = 0; i < kPlanarVoxelNum; i++)
454  {
455  planarVoxelElev[i] = 0;
456  planarVoxelEdge[i] = 0;
457  planarVoxelDyObs[i] = 0;
458  planarPointElev[i].clear();
459  }
460 
461  int terrainCloudSize = terrainCloud->points.size();
462  for (int i = 0; i < terrainCloudSize; i++)
463  {
464  point = terrainCloud->points[i];
465 
466  int indX = static_cast<int>((point.x - vehicleX + planarVoxelSize / 2) /
467  planarVoxelSize) +
468  planarVoxelHalfWidth;
469  int indY = static_cast<int>((point.y - vehicleY + planarVoxelSize / 2) /
470  planarVoxelSize) +
471  planarVoxelHalfWidth;
472 
473  if (point.x - vehicleX + planarVoxelSize / 2 < 0)
474  indX--;
475  if (point.y - vehicleY + planarVoxelSize / 2 < 0)
476  indY--;
477 
478  if (point.z - vehicleZ > min_rel_z_ && point.z - vehicleZ < max_rel_z_)
479  {
480  for (int dX = -1; dX <= 1; dX++)
481  {
482  for (int dY = -1; dY <= 1; dY++)
483  {
484  if (indX + dX >= 0 && indX + dX < planarVoxelWidth &&
485  indY + dY >= 0 && indY + dY < planarVoxelWidth)
486  {
487  planarPointElev[planarVoxelWidth * (indX + dX) + indY + dY]
488  .push_back(point.z);
489  }
490  }
491  }
492  }
493 
494  if (clear_dynamic_obs_)
495  {
496  if (indX >= 0 && indX < planarVoxelWidth && indY >= 0 &&
497  indY < planarVoxelWidth)
498  {
499  float pointX1 = point.x - vehicleX;
500  float pointY1 = point.y - vehicleY;
501  float pointZ1 = point.z - vehicleZ;
502 
503  float dis1 = sqrt(pointX1 * pointX1 + pointY1 * pointY1);
504  if (dis1 > min_dy_obs_dis_)
505  {
506  float angle1 = atan2(pointZ1 - min_dy_obs_rel_z_, dis1) * 180.0 / M_PI;
507  if (angle1 > min_dy_obs_angle_)
508  {
509  float pointX2 =
510  pointX1 * cosVehicleYaw + pointY1 * sinVehicleYaw;
511  float pointY2 =
512  -pointX1 * sinVehicleYaw + pointY1 * cosVehicleYaw;
513  float pointZ2 = pointZ1;
514 
515  float pointX3 =
516  pointX2 * cosVehiclePitch - pointZ2 * sinVehiclePitch;
517  float pointY3 = pointY2;
518  float pointZ3 =
519  pointX2 * sinVehiclePitch + pointZ2 * cosVehiclePitch;
520 
521  float pointX4 = pointX3;
522  float pointY4 =
523  pointY3 * cosVehicleRoll + pointZ3 * sinVehicleRoll;
524  float pointZ4 =
525  -pointY3 * sinVehicleRoll + pointZ3 * cosVehicleRoll;
526 
527  float dis4 = sqrt(pointX4 * pointX4 + pointY4 * pointY4);
528  float angle4 = atan2(pointZ4, dis4) * 180.0 / M_PI;
529  if ((angle4 > min_dy_obs_vfov_ && angle4 < max_dy_obs_vfov_) ||
530  fabs(pointZ4) < abs_dy_obs_rel_z_thre_)
531  {
532  planarVoxelDyObs[planarVoxelWidth * indX + indY]++;
533  }
534  }
535  }
536  else
537  {
538  planarVoxelDyObs[planarVoxelWidth * indX + indY] +=
539  min_dy_obs_point_num_;
540  }
541  }
542  }
543  }
544 
545  if (clear_dynamic_obs_)
546  {
547  for (int i = 0; i < laserCloudCropSize; i++)
548  {
549  point = laserCloudCrop->points[i];
550 
551  int indX =
